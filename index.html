The jQuery plugin for equal height of the elements in the row

How to Use

Call the plugin on the elements to equal their height. 
            $('.block').equalHeightInRow();
Equal height child elements 
            $('.block').equalHeightInRow({
                child: ['.inner-block','.inner-block-2']
            });
Callback's 
        $('.block').equalHeightInRow({

            // Before assigning height in the row
            onRowBefore: function ( elements ) {
                console.log( 'elements in row before set height', elements )
            },
            // After assigning height in the row
            onRowAfter: function ( elements ) {
                console.log( 'elements in row after set height', elements )
            },
            // Before the resize event
            onResizeBefore: function ( elements ) {
                console.log( 'all elements before resize', elements )
            },
            // After the resize event
            onResizeAfter: function ( elements ) {
                console.log( 'all elements after resize', elements )
            },
            // Executes immediately after plugin is fully loaded
            onLoad: function ( elements ) {
                console.log( 'all elements after set height', elements )
            }
        });
